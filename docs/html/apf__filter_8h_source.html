156  double max(-1.0/0.0);
157  for(size_t ii = 0; ii < N; ++ii)
158  {
159  // sample particles
160  m_particles[ii] = q1Samp(data);
161  m_logUnNormWeights[ii] = logMuEv(m_particles[ii]);
162  m_logUnNormWeights[ii] += logGEv(data, m_particles[ii]);
163  m_logUnNormWeights[ii] -= logQ1Ev(m_particles[ii], data);
164 
165  // update maximum
166  if( m_logUnNormWeights[ii] > max)
167  max = m_logUnNormWeights[ii];
168  }
169 
170  // calculate log-likelihood with log-exp-sum trick
171  double sumExp(0.0);
172  for( size_t i = 0; i < N; ++i){
173  sumExp += std::exp( m_logUnNormWeights[i] - max );
174  }
175  m_logLastCondLike = - std::log( static_cast<double>(N) ) + max + std::log(sumExp);
176 
177  // calculate expectations before you resample
178  m_expectations.resize(fs.size());
179  m_dimState = q1Samp(data).rows();
180  std::fill(m_expectations.begin(), m_expectations.end(), Vec::Zero(m_dimState)); // TODO: should this be Mat::Zero(m_dimState, m_dimState)?
181  int fId(0);
182  double weightNormConst;
183  for(auto & h : fs){
184  weightNormConst = 0.0;
185  for(size_t prtcl = 0; prtcl < N; ++prtcl){ // iterate over all particles
186  m_expectations[fId] += h(m_particles[prtcl]) * std::exp(m_logUnNormWeights[prtcl] - max);
187  weightNormConst += std::exp(m_logUnNormWeights[prtcl] - max);
188  }
189  m_expectations[fId] /= weightNormConst;
190  fId++;
191  }

194  if (m_resampTechnique == APFResampStyle::everytime_multinomial)

197  // advance time step

199  }

216

219  m2 = logFirstStageUnNormWeights[ii];
220 
221  }
222

224  std::array<unsigned int, N> myKs;
225  kGen(logFirstStageUnNormWeights, myKs);
226 
227  // now draw xts
228  double m1(-1.0/0.0);
229  double first_cll_sum(0.0);
230  double second_cll_sum(0.0);
231  double third_cll_sum(0.0);

235  second_cll_sum += std::exp( logFirstStageUnNormWeights[ii] - m2 );
236  third_cll_sum += std::exp( m_logUnNormWeights[ii] - m3 );
